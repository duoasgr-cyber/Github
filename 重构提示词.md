## 一、全局重构原则

1. **优先复用**：不重写已有模块，而是扩展/改造它们
2. **最小改动**：每个改动尽量只修改 1-2 个文件
3. **信号驱动**：沿用现有的 PyQt5 信号槽机制
4. **数据兼容**：`workflows.json` 的现有格式保持向后兼容，新增字段用可选字段
5. **渐进式**：按 Phase 顺序执行，每阶段可独立验证

### 2.2 Task 模型设计

**新增文件：`core/task_manager.py`**

```python
class Task:
    """一个任务 = 一个独立的执行上下文"""
    def __init__(self, task_id: str, name: str):
        self.id = task_id              # UUID
        self.name = name               # 显示名称，如"任务一"
        self.device_serial = None      # 绑定的设备序列号（持久化）
        self.device_alias = ""         # 用户自定义设备名称
        self.current_workflow = ""     # 当前选中的工作流名称
        self.workflow_view_state = {   # 步骤列表的展开/折叠状态
            "expanded_groups": [],     # 展开的大栏目索引
            "selected_step": -1,       # 当前选中步骤
            "scroll_position": 0,      # 滚动位置
        }
        self.ui_snapshot = {           # UI状态快照（导航选中项等）
            "current_panel": "workflow_editor",
        }
        self.created_at = ""           # 创建时间
        self.modified_at = ""          # 最后修改时间

class TaskManager(QObject):
    """管理所有任务的生命周期"""
    task_created = pyqtSignal(str)     # task_id
    task_switched = pyqtSignal(str)    # task_id
    task_deleted = pyqtSignal(str)     # task_id
    task_saved = pyqtSignal(str)       # task_id
    save_requested = pyqtSignal(str)   # 请求保存确认

    def __init__(self, base_dir, parent=None): ...
    def create_task(self, name=None) -> Task: ...        # 自动命名"任务N"
    def switch_task(self, task_id: str) -> bool: ...     # 切换前发送save_requested
    def delete_task(self, task_id: str) -> bool: ...     # 至少保留一个
    def save_task(self, task_id: str) -> None: ...       # 持久化到JSON
    def load_tasks(self) -> None: ...                    # 启动时加载
    def get_current_task(self) -> Task: ...
    def get_all_tasks(self) -> List[Task]: ...
```

# main_window.py
def _on_task_switch_requested(self, new_task_id):
    current = self._task_manager.get_current_task()
    if self._has_unsaved_changes(current):
        reply = QMessageBox.question(
            self, "保存确认",
            f"是否保存当前任务「{current.name}」的更改？",
            QMessageBox.Save | QMessageBox.Discard | QMessageBox.Cancel
        )
        if reply == QMessageBox.Save:
            self._save_current_task()
        elif reply == QMessageBox.Cancel:
            return  # 取消切换
    self._task_manager.switch_task(new_task_id)

def _save_task_state(self, task: Task):

### 9.1 variable（变量处理）

**修改文件：`core/step_executor.py`**

```python
# 在 __init__ 中新增
self._variables: dict = {}  # 运行时变量存储

# 新增方法
def _step_variable(self, step: dict) -> bool:
    var_name = step.get("variable_name", "")
    mode = step.get("mode", "boolean")
    if mode == "boolean":
        self._variables[var_name] = step.get("boolean_value", True)
    elif mode == "integer":
        current = self._variables.get(var_name, 0)
        op = step.get("integer_op", "set")
        value = step.get("integer_value", 0)
        if op == "set":
            self._variables[var_name] = value
        elif op == "add":
            self._variables[var_name] = current + value
        elif op == "sub":
            self._variables[var_name] = current - value
        elif op == "mul":
            self._variables[var_name] = current * value
        elif op == "div":
            self._variables[var_name] = current / value if value != 0 else current
    elif mode == "string":
        self._variables[var_name] = step.get("string_value", "")
    logger.info("变量操作: %s = %s", var_name, self._variables[var_name])
    return True
```

def _step_adb_command(self, step: dict) -> bool:
    command = step.get("command", "")
    if not command:
        return False
    result = self._adb_core.shell(command)
    if step.get("result_variable"):
        self._variables[step["result_variable"]] = result
    return True
