### 改动 1：新建 `core/task_manager.py`

创建 Task 模型和 TaskManager，负责任务的创建/切换/删除/持久化。

```python
import json
import os
import uuid
import logging
from datetime import datetime
from typing import Optional, List, Dict

from PyQt5.QtCore import QObject, pyqtSignal

logger = logging.getLogger(__name__)


class Task:
    """一个任务 = 一个独立执行上下文（设备+方案+UI状态）"""

    def __init__(self, task_id: str = "", name: str = ""):
        self.id = task_id or str(uuid.uuid4())
        self.name = name or "未命名任务"
        self.device_serial: str = ""
        self.device_alias: str = ""
        self.current_workflow: str = ""
        self.selected_step_index: int = -1
        self.step_scroll_position: int = 0
        self.group_mode: bool = True
        self.created_at: str = datetime.now().isoformat()
        self.modified_at: str = datetime.now().isoformat()

    def to_dict(self) -> dict:
        return {
            "name": self.name,
            "device_serial": self.device_serial,
            "device_alias": self.device_alias,
            "current_workflow": self.current_workflow,
            "selected_step_index": self.selected_step_index,
            "step_scroll_position": self.step_scroll_position,
            "group_mode": self.group_mode,
            "created_at": self.created_at,
            "modified_at": self.modified_at,
        }

    @staticmethod
    def from_dict(task_id: str, data: dict) -> "Task":
        t = Task(task_id=task_id)
        t.name = data.get("name", "未命名任务")
        t.device_serial = data.get("device_serial", "")
        t.device_alias = data.get("device_alias", "")
        t.current_workflow = data.get("current_workflow", "")
        t.selected_step_index = data.get("selected_step_index", -1)
        t.step_scroll_position = data.get("step_scroll_position", 0)
        t.group_mode = data.get("group_mode", True)
        t.created_at = data.get("created_at", "")
        t.modified_at = data.get("modified_at", "")
        return t


class TaskManager(QObject):
    """管理所有任务的生命周期和持久化"""

    task_created = pyqtSignal(str)
    task_switched = pyqtSignal(str)
    task_deleted = pyqtSignal(str)
    task_saved = pyqtSignal(str)

    def __init__(self, base_dir: str, parent=None):
        super().__init__(parent)
        self._tasks_path = os.path.join(base_dir, "config", "tasks.json")
        self._tasks: Dict[str, Task] = {}
        self._current_task_id: str = ""
        self._counter: int = 0

    def load_tasks(self):
        if not os.path.exists(self._tasks_path):
            return
        try:
            with open(self._tasks_path, "r", encoding="utf-8") as f:
                data = json.load(f)
            self._current_task_id = data.get("current_task_id", "")
            for tid, tdata in data.get("tasks", {}).items():
                self._tasks[tid] = Task.from_dict(tid, tdata)
            self._counter = len(self._tasks)
        except Exception as e:
            logger.error("加载任务失败: %s", e)

    def save_all(self):
        data = {
            "current_task_id": self._current_task_id,
            "tasks": {tid: t.to_dict() for tid, t in self._tasks.items()},
        }
        os.makedirs(os.path.dirname(self._tasks_path), exist_ok=True)
        tmp = self._tasks_path + ".tmp"
        with open(tmp, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        import shutil
        shutil.move(tmp, self._tasks_path)

    def save_task(self, task_id: str):
        if task_id in self._tasks:
            self._tasks[task_id].modified_at = datetime.now().isoformat()
        self.save_all()
        self.task_saved.emit(task_id)

    def create_task(self, name: str = "") -> Task:
        self._counter += 1
        if not name:
            name = f"任务{self._counter}"
        task = Task(name=name)
        self._tasks[task.id] = task
        self._current_task_id = task.id
        self.save_all()
        self.task_created.emit(task.id)
        return task

    def switch_task(self, task_id: str) -> bool:
        if task_id not in self._tasks:
            return False
        self._current_task_id = task_id
        self.save_all()
        self.task_switched.emit(task_id)
        return True

    def delete_task(self, task_id: str) -> bool:
        if task_id not in self._tasks:
            return False
        if len(self._tasks) <= 1:
            return False
        del self._tasks[task_id]
        if self._current_task_id == task_id:
            self._current_task_id = next(iter(self._tasks))
        self.save_all()
        self.task_deleted.emit(task_id)
        return True

    def get_current_task(self) -> Optional[Task]:
        return self._tasks.get(self._current_task_id)

    def get_all_tasks(self) -> List[Task]:
        return list(self._tasks.values())

    def get_task_by_id(self, task_id: str) -> Optional[Task]:
        return self._tasks.get(task_id)
```

from PyQt5.QtCore import Qt, pyqtSignal
from PyQt5.QtGui import QFont
from PyQt5.QtWidgets import QWidget, QHBoxLayout, QPushButton, QTabBar

        self._btn_add.setFixedSize(32, 32)
        self._btn_add.setToolTip("新建任务")
        self._btn_add.setFont(QFont("Segoe UI", 16, QFont.Bold))
        self._btn_add.clicked.connect(self.create_task_requested.emit)
        layout.addWidget(self._btn_add)

    def add_task_tab(self, task_id: str, name: str):
        self._task_ids.append(task_id)
        self._tab_bar.addTab(name)
        self._tab_bar.setCurrentIndex(len(self._task_ids) - 1)

    def remove_task_tab(self, task_id: str):
        if task_id in self._task_ids:
            idx = self._task_ids.index(task_id)
            self._task_ids.pop(idx)
            self._tab_bar.removeTab(idx)

    def set_current_task(self, task_id: str):
        if task_id in self._task_ids:
            idx = self._task_ids.index(task_id)
            self._tab_bar.blockSignals(True)
            self._tab_bar.setCurrentIndex(idx)
            self._tab_bar.blockSignals(False)

    def update_task_name(self, task_id: str, name: str):
        if task_id in self._task_ids:
            self._tab_bar.setTabText(self._task_ids.index(task_id), name)

    def _on_tab_changed(self, index: int):
        if 0 <= index < len(self._task_ids):
            self.task_switch_requested.emit(self._task_ids[index])

    def _on_tab_close(self, index: int):
        if 0 <= index < len(self._task_ids):
            self.task_close_requested.emit(self._task_ids[index])

        item = self._table.item(row, 1)

            self._selected_serial = item.text()
            self.accept()

        if name in workflows:

        else:
            wf = {"description": "", "device_resolution": {"width": 2400, "height": 1080},
                  "steps": []}

            QMessageBox.information(self, "提示", "请先选择一个方案")
            return
        workflows = self._config_manager.get_all_workflows()
        if len(workflows) <= 1:
            QMessageBox.warning(self, "提示", "不能删除最后一个方案")
            return
        reply = QMessageBox.question(self, "确认删除", f"确定要删除方案 '{name}' 吗？",
                                     QMessageBox.Yes | QMessageBox.No, QMessageBox.No)
        if reply == QMessageBox.Yes:
            self._config_manager.delete_workflow(name)
            self._refresh_list()
            self.workflows_changed.emit()

            return

        if not ok or not new_name.strip() or new_name.strip() == name:
            return

            with open(file_path, "r", encoding="utf-8") as f:
                data = json.load(f)
            if "workflows" in data:
                for wf_name, wf_data in data["workflows"].items():
                    self._config_manager.set_workflow(wf_name, wf_data)
            elif "steps" in data:
                base = os.path.splitext(os.path.basename(file_path))[0]
                self._config_manager.set_workflow(base, data)
            self._refresh_list()
            self.workflows_changed.emit()
            QMessageBox.information(self, "导入成功", "方案已导入")

            path = os.path.join(target, f"{name}.json")
            with open(path, "w", encoding="utf-8") as f:
                json.dump(wf, f, ensure_ascii=False, indent=2)
            QMessageBox.information(self, "导出成功", f"已导出到:\n{path}")
        except Exception as e:
            QMessageBox.critical(self, "导出失败", str(e))

```python
class StepListWidget(QListWidget):
    step_order_changed = pyqtSignal()
    step_clicked = pyqtSignal(int)

    def __init__(self, parent=None):
        super().__init__(parent)
        self.setFont(QFont("Microsoft YaHei", 10))
        self.setDragDropMode(QAbstractItemView.InternalMove)
        self.setDefaultDropAction(Qt.MoveAction)
        self.currentRowChanged.connect(self._on_row_changed)
        self._group_mode: bool = True
        self._raw_steps: list = []

    def toggle_group_mode(self):
        self._group_mode = not self._group_mode
        self.load_steps(self._raw_steps)

    def load_steps(self, steps: list):
        self._raw_steps = steps
        self.blockSignals(True)
        self.clear()
        if self._group_mode:
            self._load_grouped(steps)
        else:
            self._load_flat(steps)
        self.blockSignals(False)

    def _load_flat(self, steps: list):
        for i, step in enumerate(steps):
            step_type = step.get("type", "unknown")
            comment = step.get("comment", "")
            display_name = step.get("display_name", "")
            type_display = STEP_TYPE_DISPLAY.get(step_type, step_type)
            text = f"{i + 1}. {display_name}" if display_name else f"{i + 1}. {type_display}"
            if not display_name and comment:
                text += f" - {comment}"
            self.addItem(QListWidgetItem(text))

    def _load_grouped(self, steps: list):
        i = 0
        group_num = 0
        while i < len(steps):
            step = steps[i]
            step_type = step.get("type", "unknown")
            comment = step.get("comment", "")
            display_name = step.get("display_name", "")

            if step_type in ("condition", "loop", "call_workflow"):
                group_num += 1
                label = display_name or comment or STEP_TYPE_DISPLAY.get(step_type, step_type)
                sub = 0
                if step_type == "loop":
                    sub = len(step.get("steps", []))
                elif step_type == "condition":
                    sub = len(step.get("then_steps", [])) + len(step.get("else_steps", []))
                suffix = f" ({sub}步)" if sub > 0 else ""
                item = QListWidgetItem(f"[{group_num}] {label}{suffix}")
                item.setData(Qt.UserRole, i)
                self.addItem(item)
                i += 1
                continue

            group_steps = []
            while i < len(steps):
                s = steps[i]
                t = s.get("type", "")
                if t in ("condition", "loop", "call_workflow"):
                    break
                group_steps.append((i, s))
                i += 1
                if len(group_steps) >= 5:
                    break

            group_num += 1
            if len(group_steps) == 1:
                idx, s = group_steps[0]
                dn = s.get("display_name", "")
                cm = s.get("comment", "")
                t = s.get("type", "unknown")
                label = dn or cm or STEP_TYPE_DISPLAY.get(t, t)
                item = QListWidgetItem(f"[{group_num}] {label}")
                item.setData(Qt.UserRole, idx)
            else:
                first = group_steps[0][1]
                dn = first.get("display_name", "")
                cm = first.get("comment", "")
                t = first.get("type", "unknown")
                label = dn or cm or STEP_TYPE_DISPLAY.get(t, "")
                item = QListWidgetItem(f"[{group_num}] {label} ... (+{len(group_steps)-1}步)")
                item.setData(Qt.UserRole, group_steps[0][0])
            self.addItem(item)

    # 保留原有的 dropEvent、get_current_index、_on_row_changed 不变

def _step_variable(self, step: dict) -> bool:
    var_name = step.get("variable_name", "")
    if not var_name:
        return False
    mode = step.get("mode", "boolean")
    if mode == "boolean":
        self._variables[var_name] = step.get("boolean_value", True)
    elif mode == "integer":
        current = self._variables.get(var_name, 0)
        if not isinstance(current, (int, float)):
            try:
                current = int(current)
            except (ValueError, TypeError):
                current = 0
        op = step.get("integer_op", "set")
        value = step.get("integer_value", 0)
        ops = {"set": lambda: value, "add": lambda: current + value,
               "sub": lambda: current - value, "mul": lambda: current * value,
               "div": lambda: current / value if value != 0 else current}
        self._variables[var_name] = ops.get(op, ops["set"])()
    elif mode == "string":
        self._variables[var_name] = step.get("string_value", "")
    else:
        return False
    return True

def _step_adb_command(self, step: dict) -> bool:
    command = step.get("command", "")
    if not command:
        return False
    try:
        result = self._adb_core.shell(command)
        rv = step.get("result_variable", "")
        if rv:
            self._variables[rv] = result
        return True
    except Exception as e:
        logger.error("ADB命令执行失败: %s - %s", command, e)
        return False

def get_variable(self, name: str, default=None):
    return self._variables.get(name, default)

def set_variable(self, name: str, value):
    self._variables[name] = value

def clear_variables(self):
    self._variables.clear()
```
